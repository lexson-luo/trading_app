Step 1 — Install system dependencies (once):

sudo dnf install -y cmake gcc-c++ ninja-build \
    libX11-devel libXext-devel libXrandr-devel libXi-devel \
    libXcursor-devel libXinerama-devel libXfixes-devel \
    wayland-devel libxkbcommon-devel wayland-protocols-devel \
    mesa-libGL-devel mesa-libEGL-devel glew-devel openssl-devel git


Step 2 — Run:

# Terminal 1 — backend
./setup.sh server-deps    # first time only
./setup.sh server-build   # first time only
./setup.sh server         # start the backend — leave this running

# Terminal 2 — desktop client

cmake -S . -B build -DCMAKE_BUILD_TYPE=Debug -DBUILD_FRONTEND=ON && cmake --build build --target hf_client -j$(nproc)

cmake --build build --target hf_client -j$(nproc)
./build/frontend/hf_client

./setup.sh client
